35template<class Basic2TThermo, class MixtureType>
36void Foam::he2TThermo<Basic2TThermo, MixtureType>::
37heBoundaryCorrection(volScalarField& h)
38{
39 volScalarField::Boundary& hBf = h.boundaryFieldRef();
40
41 forAll(hBf, patchi)
42 {
43 if (isA<gradientEnergyFvPatchScalarField>(hBf[patchi]))
44 {
45 refCast<gradientEnergyFvPatchScalarField>(hBf[patchi]).gradient()
46 = hBf[patchi].fvPatchField::snGrad();
47 }
48 else if (isA<mixedEnergyFvPatchScalarField>(hBf[patchi]))
49 {
50 refCast<mixedEnergyFvPatchScalarField>(hBf[patchi]).refGrad()
51 = hBf[patchi].fvPatchField::snGrad();
52 }
53 }
54}

57template<class Basic2TThermo, class MixtureType>
58void Foam::he2TThermo<Basic2TThermo, MixtureType>::init
59(
60 const volScalarField& p,
61 const volScalarField& TTR,
62 const volScalarField& TVib,
63 volScalarField& he
64)
65{
66 scalarField& heCells = he.primitiveFieldRef();
67 const scalarField& pCells = p.primitiveField();
68 const scalarField& TTRCells = TTR.primitiveField();
69 //const scalarField& TVibCells = TVib.primitiveField();
70
71 forAll(heCells, celli)
72 {
73 heCells[celli] =
74 this->cellMixture(celli).HE(pCells[celli], TTRCells[celli]);
75 }
76
77 volScalarField::Boundary& heBf = he.boundaryFieldRef();
78
79 forAll(heBf, patchi)
80 {
81 heBf[patchi] == this->he
82 (
83 p.boundaryField()[patchi],
84 TTR.boundaryField()[patchi],
85 patchi
86 );
87
88 heBf[patchi].useImplicit(TTR.boundaryField()[patchi].useImplicit());
89 }
90
91 this->heBoundaryCorrection(he);
92
93 // Note: TTR does not have oldTime
94 if (p.nOldTimes())
95 {
96 init(p.oldTime(), TTR.oldTime(), TVib.oldTime(), he.oldTime());
97 }
98}

214template<class Basic2TThermo, class MixtureType>
215Foam::tmp<Foam::volScalarField> Foam::he2TThermo<Basic2TThermo, MixtureType>::he
216(
217 const volScalarField& p,
218 const volScalarField& TTR
219) const
220{
221 const fvMesh& mesh = this->TTR_.mesh();
222
223 auto the = volScalarField::New
224 (
225 "he",
226 IOobject::NO_REGISTER,
227 mesh,
228 he_.dimensions()
229 );
230 auto& he = the.ref();
231
232 scalarField& heCells = he.primitiveFieldRef();
233 const scalarField& pCells = p;
234 const scalarField& TTRCells = TTR;
235
236 forAll(heCells, celli)
237 {
238 heCells[celli] =
239 this->cellMixture(celli).HE(pCells[celli], TTRCells[celli]);
240 }
241
242 volScalarField::Boundary& heBf = he.boundaryFieldRef();
243
244 forAll(heBf, patchi)
245 {
246 scalarField& hep = heBf[patchi];
247 const scalarField& pp = p.boundaryField()[patchi];
248 const scalarField& TTRp = TTR.boundaryField()[patchi];
249
250 forAll(hep, facei)
251 {
252 hep[facei] =
253 this->patchFaceMixture(patchi, facei).HE(pp[facei], TTRp[facei]);
254 }
255 }
256
257 return the;
258}

772{

774
775 auto tCpByCpvTR = volScalarField::New
776 (
777 "CpByCpvTR",

781 );

783
784 forAll(this->TTR_, celli)
785 {
786 CpByCpvTR[celli] = this->cellMixture(celli).CpByCpvTR
787 (
788 this->p_[celli],
789 this->TTR_[celli]
790 );
791 }
792
793 volScalarField::Boundary& CpByCpvTRBf =
794 CpByCpvTR.boundaryFieldRef();
795
796 forAll(CpByCpvTRBf, patchi)
797 {
798 const fvPatchScalarField& pp = this->p_.boundaryField()[patchi];
799 const fvPatchScalarField& pTTR = this->TTR_.boundaryField()[patchi];
800 fvPatchScalarField& pCpByCpvTR = CpByCpvTRBf[patchi];
801
802 forAll(pTTR, facei)
803 {
