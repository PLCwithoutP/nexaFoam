33template<class EquationOfState>
34void Foam::janaf2TThermo<EquationOfState>::checkInputData() const
35{
36 if (Tlow_ >= Thigh_)
37 {
38 FatalErrorInFunction
39 << "Tlow(" << Tlow_ << ") >= Thigh(" << Thigh_ << ')'
40 << exit(FatalError);
41 }
42
43 if (Tcommon_ <= Tlow_)
44 {
45 FatalErrorInFunction
46 << "Tcommon(" << Tcommon_ << ") <= Tlow(" << Tlow_ << ')'
47 << exit(FatalError);
48 }
49
50 if (Tcommon_ > Thigh_)
51 {
52 FatalErrorInFunction
53 << "Tcommon(" << Tcommon_ << ") > Thigh(" << Thigh_ << ')'
54 << exit(FatalError);
55 }
56}

61template<class EquationOfState>
62Foam::janaf2TThermo<EquationOfState>::janaf2TThermo(const dictionary& dict)
63:
64 EquationOfState(dict),
65 Tlow_(dict.subDict("thermodynamics").get<scalar>("Tlow")),
66 Thigh_(dict.subDict("thermodynamics").get<scalar>("Thigh")),
67 Tcommon_(dict.subDict("thermodynamics").get<scalar>("Tcommon")),
68 ThetaVib_(dict.subDict("thermodynamics").get<scalar>("ThetaVibrational")),
69 highCpCoeffs_(dict.subDict("thermodynamics").lookup("highCpCoeffs")),
70 lowCpCoeffs_(dict.subDict("thermodynamics").lookup("lowCpCoeffs"))
71{
72 // Convert coefficients to mass-basis
73 for (label coefLabel=0; coefLabel<nCoeffs_; coefLabel++)
74 {
75 highCpCoeffs_[coefLabel] *= this->R();
76 lowCpCoeffs_[coefLabel] *= this->R();
77 }
78
79 checkInputData();
80}

85template<class EquationOfState>
86void Foam::janaf2TThermo<EquationOfState>::write(Ostream& os) const
87{
88 EquationOfState::write(os);
89
90 // Convert coefficients back to dimensionless form
91 coeffArray highCpCoeffs;
92 coeffArray lowCpCoeffs;
93 for (label coefLabel=0; coefLabel<nCoeffs_; coefLabel++)
94 {
95 highCpCoeffs[coefLabel] = highCpCoeffs_[coefLabel]/this->R();
96 lowCpCoeffs[coefLabel] = lowCpCoeffs_[coefLabel]/this->R();
97 }
98
99 // Entries in dictionary format
100 {
101 os.beginBlock("thermodynamics");
102 os.writeEntry("Tlow", Tlow_);
103 os.writeEntry("Thigh", Thigh_);
104 os.writeEntry("Tcommon", Tcommon_);
105 os.writeEntry("ThetaVib", ThetaVib_);
106 os.writeEntry("highCpCoeffs", highCpCoeffs);
107 os.writeEntry("lowCpCoeffs", lowCpCoeffs);
108 os.endBlock();
109 }
110}
